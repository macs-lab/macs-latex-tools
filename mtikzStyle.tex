% tikz plot
\newcommand{\plotfun}[3][6cm]{
    \begin{tikzpicture}
        \begin{axis}
        [width=#1,grid=major,samples=100,mark=none]
        \addplot[blue,very thick,domain=#3]
        {#2};
        \end{axis}
    \end{tikzpicture}
}
